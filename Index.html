// Function to start fireworks and play sound function startFireworks() { const sound = document.getElementById("diwaliSound"); if (sound) sound.play(); for (let i = 0; i < 8; i++) createFirework(); } function createFirework() { const firework = document.createElement("div"); firework.classList.add("firework"); document.getElementById("firework-container").appendChild(firework); firework.style.left = Math.random() * 100 + "%"; firework.style.top = Math.random() * 60 + "%"; firework.style.animationDuration = (Math.random() * 1.5 + 1) + "s"; setTimeout(() => firework.remove(), 2000); }
